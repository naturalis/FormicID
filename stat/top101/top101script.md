capitalize <- function(x) {
    first <- toupper(substr(x, start = 1, stop = 1))
    rest <- tolower(substr(x, start = 2, stop = nchar(x)))
    paste0(first, rest)
}
levels(top101_transp$scientific_name) <-

File size distribution
